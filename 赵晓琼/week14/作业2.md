# 股票可视化分析 Skill

## 功能概述

本 Skill 提供股票数据的可视化分析功能，主要特点：

1. **双周期对比分析**：将日K线和周K线数据绘制在同一图表中，方便对比短期和中期趋势
2. **趋势识别**：自动识别上升趋势、下降趋势和横盘震荡
3. **买卖信号分析**：基于均线交叉、K线形态等技术指标给出买卖建议

## 数据获取

使用 https://api.autostock.cn API 获取数据：

- `get_day_line` - 获取日K线数据
- `get_week_line` - 获取周K线数据
- `get_stock_info` - 获取股票基础信息

## 分析逻辑

### 趋势判断

| 月K方向 | 周K状态 | 日K信号 | 操作建议 |
|----------|----------|----------|-----------|
| 上升 | 回调到支撑 | 出现买入形态 | **买入或加仓** |
| 上升 | 接近压力 | 出现弱势/背离 | 减仓或观望 |
| 下降 | 反弹 | 出现卖出形态 | **反弹卖出或空仓** |
| 下降 | 震荡 | 无明确信号 | 不操作 |
| 震荡 | 震荡 | 短线信号 | 小仓位做波段 |

### 买入信号

- 日K线出现放量阳线，站上5/10日均线
- 周K线在支撑位出现下影线
- 周K线多头排列（5周均线 > 10周均线 > 20周均线）

### 卖出信号

- 日K线出现放量阴线，跌破关键支撑
- 周K线在压力位出现长上影线
- 周K线空头排列（5周均线 < 10周均线 < 20周均线）

## 实现代码

```python
TOKEN = "zgaLG8unUPr"

import requests
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from datetime import datetime, timedelta
import pandas as pd
import numpy as np
from typing import Dict, List, Optional, Tuple

# 设置中文字体支持
plt.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'SimHei', 'DejaVu Sans']
plt.rcParams['axes.unicode_minus'] = False


class StockVisualizer:
    """股票可视化分析类"""

    def __init__(self, token: str = TOKEN):
        self.token = token
        self.base_url = "https://api.autostock.cn/v1"

    def get_stock_info(self, code: str) -> Dict:
        """获取股票基础信息"""
        url = f"{self.base_url}/stock?token={self.token}&code={code}"
        try:
            response = requests.get(url, timeout=10)
            return response.json()
        except Exception as e:
            print(f"获取股票信息失败: {e}")
            return {}

    def get_day_kline(self, code: str, days: int = 90) -> Dict:
        """获取日K线数据"""
        end_date = datetime.now().strftime("%Y-%m-%d")
        start_date = (datetime.now() - timedelta(days=days)).strftime("%Y-%m-%d")

        url = f"{self.base_url}/stock/kline/day"
        params = {
            "token": self.token,
            "code": code,
            "startDate": start_date,
            "endDate": end_date,
            "type": 1  # 前复权
        }
        try:
            response = requests.get(url, params=params, timeout=10)
            return response.json()
        except Exception as e:
            print(f"获取日K线失败: {e}")
            return {}

    def get_week_kline(self, code: str, weeks: int = 52) -> Dict:
        """获取周K线数据"""
        end_date = datetime.now().strftime("%Y-%m-%d")
        start_date = (datetime.now() - timedelta(weeks=weeks)).strftime("%Y-%m-%d")

        url = f"{self.base_url}/stock/kline/week"
        params = {
            "token": self.token,
            "code": code,
            "startDate": start_date,
            "endDate": end_date,
            "type": 1  # 前复权
        }
        try:
            response = requests.get(url, params=params, timeout=10)
            return response.json()
        except Exception as e:
            print(f"获取周K线失败: {e}")
            return {}

    def calculate_ma(self, data: List[Dict], period: int) -> List[float]:
        """计算移动平均线"""
        closes = [item.get('close', 0) for item in data]
        ma = []
        for i in range(len(closes)):
            if i < period - 1:
                ma.append(None)
            else:
                ma.append(sum(closes[i-period+1:i+1]) / period)
        return ma

    def analyze_trend(self, data: List[Dict], ma_periods: List[int] = [5, 10, 20]) -> Tuple[str, Dict]:
        """
        分析趋势
        返回: (趋势方向, 均线数据字典)
        """
        if not data or len(data) < max(ma_periods):
            return "数据不足", {}

        # 计算各周期均线
        ma_data = {}
        for period in ma_periods:
            ma_data[f"MA{period}"] = self.calculate_ma(data, period)

        # 获取最新的均线值
        latest_mas = {k: v[-1] for k, v in ma_data.items() if v[-1] is not None}

        if len(latest_mas) < 3:
            return "数据不足", ma_data

        # 判断趋势
        ma5 = latest_mas.get("MA5", 0)
        ma10 = latest_mas.get("MA10", 0)
        ma20 = latest_mas.get("MA20", 0)

        if ma5 > ma10 > ma20:
            trend = "上升趋势"
        elif ma5 < ma10 < ma20:
            trend = "下降趋势"
        else:
            trend = "震荡趋势"

        return trend, ma_data

    def generate_trading_signal(self, day_data: List[Dict], week_data: List[Dict]) -> Dict:
        """
        生成交易信号
        返回包含买入/卖出建议的字典
        """
        signal = {
            "action": "观望",
            "confidence": 0,
            "reasons": [],
            "stop_loss": None,
            "take_profit": None
        }

        # 分析日K线趋势
        day_trend, day_ma = self.analyze_trend(day_data)
        # 分析周K线趋势
        week_trend, week_ma = self.analyze_trend(week_data)

        # 获取最新数据
        if not day_data or not week_data:
            return signal

        latest_day = day_data[-1]
        latest_week = week_data[-1]

        # 买入信号判断
        buy_signals = []

        # 1. 周K线趋势向上
        if week_trend == "上升趋势":
            buy_signals.append("周K线处于上升趋势")

        # 2. 日K线突破均线
        if day_trend == "上升趋势":
            buy_signals.append("日K线形成多头排列")

        # 3. 最近一根K线是阳线且放量
        if len(day_data) >= 2:
            prev_day = day_data[-2]
            if (latest_day.get('close', 0) > latest_day.get('open', 0) and
                latest_day.get('volume', 0) > prev_day.get('volume', 0)):
                buy_signals.append("日K线出现放量阳线")

        # 4. 周K线回调到支撑位（MA20附近）
        if week_ma.get("MA20") and latest_week.get('close', 0):
            ma20_week = week_ma["MA20"][-1]
            close_week = latest_week.get('close', 0)
            if ma20_week and abs(close_week - ma20_week) / ma20_week < 0.02:
                buy_signals.append("周K线回调至MA20支撑位")

        # 卖出信号判断
        sell_signals = []

        # 1. 周K线趋势向下
        if week_trend == "下降趋势":
            sell_signals.append("周K线处于下降趋势")

        # 2. 日K线跌破均线
        if day_trend == "下降趋势":
            sell_signals.append("日K线形成空头排列")

        # 3. 最近一根K线是阴线且放量
        if len(day_data) >= 2:
            prev_day = day_data[-2]
            if (latest_day.get('close', 0) < latest_day.get('open', 0) and
                latest_day.get('volume', 0) > prev_day.get('volume', 0)):
                sell_signals.append("日K线出现放量阴线")

        # 综合判断
        if len(buy_signals) >= 2:
            signal["action"] = "买入"
            signal["confidence"] = min(len(buy_signals) * 30, 90)
            signal["reasons"] = buy_signals
            # 设置止损位（最近一根标志性阳线的最低点）
            if len(day_data) >= 3:
                signal["stop_loss"] = min(
                    day_data[-1].get('low', 0),
                    day_data[-2].get('low', 0),
                    day_data[-3].get('low', 0)
                )
        elif len(sell_signals) >= 2:
            signal["action"] = "卖出"
            signal["confidence"] = min(len(sell_signals) * 30, 90)
            signal["reasons"] = sell_signals
        else:
            signal["action"] = "观望"
            signal["confidence"] = 50
            signal["reasons"] = ["趋势不明确，建议观望"]

        return signal

    def visualize_stock(self, code: str, save_path: Optional[str] = None) -> Dict:
        """
        可视化股票数据（日K和周K合并展示）

        参数:
            code: 股票代码
            save_path: 图片保存路径（可选）

        返回:
            分析结果字典
        """
        # 获取股票信息
        stock_info = self.get_stock_info(code)
        stock_name = stock_info.get('data', {}).get('name', code)

        # 获取K线数据
        day_data = self.get_day_kline(code)
        week_data = self.get_week_kline(code)

        # 解析数据
        day_list = day_data.get('data', [])
        week_list = week_data.get('data', [])

        if not day_list or not week_list:
            print(f"无法获取股票 {code} 的数据")
            return {}

        # 分析趋势
        day_trend, day_ma = self.analyze_trend(day_list)
        week_trend, week_ma = self.analyze_trend(week_list)

        # 生成交易信号
        trading_signal = self.generate_trading_signal(day_list, week_list)

        # 创建图表
        fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(16, 10), gridspec_kw={'height_ratios': [2, 1]})
        fig.suptitle(f'{stock_name} ({code}) 股价分析', fontsize=16, fontweight='bold')

        # 绘制日K线
        self._plot_kline(ax1, day_list, day_ma, "日K线")

        # 绘制周K线
        self._plot_kline(ax2, week_list, week_ma, "周K线")

        plt.tight_layout()

        # 保存或显示图表
        if save_path:
            plt.savefig(save_path, dpi=150, bbox_inches='tight')
            print(f"图表已保存至: {save_path}")
        else:
            plt.show()

        plt.close()

        # 返回分析结果
        result = {
            "code": code,
            "name": stock_name,
            "day_trend": day_trend,
            "week_trend": week_trend,
            "trading_signal": trading_signal,
            "latest_price": day_list[-1].get('close', 0) if day_list else 0,
            "analysis_date": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }

        return result

    def _plot_kline(self, ax, data: List[Dict], ma_data: Dict, title: str):
        """绘制K线图（简化版，使用折线图）"""
        dates = []
        closes = []
        highs = []
        lows = []
        opens = []

        for item in data:
            dates.append(datetime.strptime(item.get('date', ''), '%Y-%m-%d'))
            closes.append(item.get('close', 0))
            highs.append(item.get('high', 0))
            lows.append(item.get('low', 0))
            opens.append(item.get('open', 0))

        # 绘制K线（使用高低点和收盘价）
        ax.fill_between(dates, lows, highs, alpha=0.3, color='gray', label='价格区间')
        ax.plot(dates, closes, color='blue', linewidth=1.5, label='收盘价')

        # 绘制均线
        colors = ['red', 'orange', 'green']
        for idx, (ma_name, ma_values) in enumerate(ma_data.items()):
            if ma_values:
                valid_dates = dates[len(dates)-len(ma_values):]
                valid_ma = [v for v in ma_values if v is not None]
                if valid_ma:
                    ax.plot(valid_dates[-len(valid_ma):], valid_ma,
                           color=colors[idx % len(colors)],
                           linewidth=1,
                           label=ma_name)

        ax.set_title(title, fontsize=12)
        ax.set_xlabel('日期')
        ax.set_ylabel('价格')
        ax.legend(loc='upper left')
        ax.grid(True, alpha=0.3)
        ax.xaxis.set_major_formatter(mdates.DateFormatter('%m-%d'))
        ax.xaxis.set_major_locator(mdates.WeekdayLocator(interval=2))
        plt.setp(ax.xaxis.get_majorticklabels(), rotation=45)


# 使用示例
if __name__ == "__main__":
    visualizer = StockVisualizer()

    # 分析股票（例如：贵州茅台 600519）
    result = visualizer.visualize_stock("600519")

    if result:
        print("\n" + "="*50)
        print(f"股票: {result['name']} ({result['code']})")
        print(f"最新价格: {result['latest_price']}")
        print(f"日K趋势: {result['day_trend']}")
        print(f"周K趋势: {result['week_trend']}")
        print(f"\n交易建议: {result['trading_signal']['action']}")
        print(f"信号强度: {result['trading_signal']['confidence']}%")
        print(f"理由: {', '.join(result['trading_signal']['reasons'])}")
        if result['trading_signal']['stop_loss']:
            print(f"建议止损位: {result['trading_signal']['stop_loss']}")
        print("="*50)
```
