# -*- coding: utf-8 -*-
"""
作业2：股票可视化Skill
功能：股票周波动和日波动可视化，并提供买入卖出建议
"""

import requests
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from datetime import datetime, timedelta
import json
import pandas as pd
import numpy as np
from typing import Dict, Tuple, List


class StockVisualizationSkill:
    """
    股票可视化Skill
    """

    def __init__(self):
        # 使用免费股票API
        self.api_base = "https://api.autostock.cn"
        plt.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'SimHei']
        plt.rcParams['axes.unicode_minus'] = False

    def get_stock_data(self, stock_code: str, days: int = 30) -> Dict:
        """
        获取股票数据

        Args:
            stock_code: 股票代码（如：000001）
            days: 获取最近多少天的数据

        Returns:
            股票数据字典
        """
        try:
            # 尝试获取日线数据
            url = f"{self.api_base}/stock/quote"
            params = {
                "code": stock_code,
                "period": "daily",
                "count": days
            }

            response = requests.get(url, params=params, timeout=10)

            if response.status_code == 200:
                data = response.json()
                if data.get('code') == 0 and data.get('data'):
                    return data['data']

            print(f"警告: API返回异常数据，使用模拟数据")
            return self._generate_mock_data(stock_code, days)

        except Exception as e:
            print(f"获取股票数据失败: {e}")
            return self._generate_mock_data(stock_code, days)

    def _generate_mock_data(self, stock_code: str, days: int) -> Dict:
        """
        生成模拟股票数据（当API不可用时使用）
        """
        print(f"生成股票 {stock_code} 的模拟数据")

        end_date = datetime.now()
        dates = [(end_date - timedelta(days=i)).strftime('%Y-%m-%d') for i in range(days-1, -1, -1)]

        # 模拟股价波动
        base_price = np.random.uniform(10, 100)
        price_changes = np.random.normal(0, 0.03, days).cumsum()
        prices = base_price * (1 + price_changes)

        # 确保价格为正数
        prices = np.maximum(prices, 1)

        # 生成开盘价、最高价、最低价
        daily_volatility = np.random.uniform(0.01, 0.05, days)
        open_prices = prices * (1 + np.random.uniform(-0.01, 0.01, days))
        high_prices = prices * (1 + daily_volatility)
        low_prices = prices * (1 - daily_volatility)

        # 计算成交量
        volumes = np.random.uniform(1000000, 10000000, days)

        data_list = []
        for i in range(days):
            data_list.append({
                'date': dates[i],
                'open': round(open_prices[i], 2),
                'high': round(high_prices[i], 2),
                'low': round(low_prices[i], 2),
                'close': round(prices[i], 2),
                'volume': int(volumes[i]),
                'change': round(prices[i] - prices[i-1] if i > 0 else 0, 2),
                'change_percent': round((prices[i] - prices[i-1]) / prices[i-1] * 100 if i > 0 else 0, 2)
            })

        return {
            'code': stock_code,
            'name': f'模拟股票{stock_code}',
            'data': data_list
        }

    def calculate_volatility(self, data: List[Dict]) -> Tuple[List, List]:
        """
        计算日波动和周波动

        Args:
            data: 股票数据列表

        Returns:
            (日波动列表, 周波动列表)
        """
        daily_volatility = []
        weekly_volatility = []

        # 计算日波动（基于高低价）
        for item in data:
            high = item['high']
            low = item['low']
            volatility = (high - low) / low * 100
            daily_volatility.append(volatility)

        # 计算周波动
        weekly_groups = []
        current_week = []

        # 按周分组
        for i in range(len(data)):
            date_str = data[i]['date']
            date = datetime.strptime(date_str, '%Y-%m-%d')
            week_num = date.isocalendar()[1]

            if not current_week:
                current_week = [{'date': date, 'data': data[i]}]
            else:
                prev_week_num = current_week[-1]['date'].isocalendar()[1]
                if week_num == prev_week_num:
                    current_week.append({'date': date, 'data': data[i]})
                else:
                    weekly_groups.append(current_week)
                    current_week = [{'date': date, 'data': data[i]}]

        if current_week:
            weekly_groups.append(current_week)

        # 计算每周的波动
        for week_data in weekly_groups:
            if len(week_data) > 0:
                week_high = max(item['data']['high'] for item in week_data)
                week_low = min(item['data']['low'] for item in week_data)
                week_volatility = (week_high - week_low) / week_low * 100
                # 扩展周波动到该周的每一天
                for _ in range(len(week_data)):
                    weekly_volatility.append(week_volatility)

        # 确保长度一致
        min_len = min(len(daily_volatility), len(weekly_volatility))
        daily_volatility = daily_volatility[:min_len]
        weekly_volatility = weekly_volatility[:min_len]

        return daily_volatility, weekly_volatility

    def generate_trading_recommendation(self, data: List[Dict], daily_vol: List, weekly_vol: List) -> Dict:
        """
        生成买入卖出建议

        Args:
            data: 股票数据
            daily_vol: 日波动列表
            weekly_vol: 周波动列表

        Returns:
            交易建议字典
        """
        latest_data = data[-1]
        latest_price = latest_data['close']
        latest_daily_vol = daily_vol[-1]
        latest_weekly_vol = weekly_vol[-1]

        avg_daily_vol = np.mean(daily_vol)
        avg_weekly_vol = np.mean(weekly_vol)

        # 计算最近几天的价格趋势
        recent_prices = [item['close'] for item in data[-5:]]
        price_trend = (recent_prices[-1] - recent_prices[0]) / recent_prices[0] * 100

        # 生成建议
        recommendation = {
            'current_price': latest_price,
            'price_trend': price_trend,
            'daily_volatility': latest_daily_vol,
            'weekly_volatility': latest_weekly_vol,
            'avg_daily_volatility': avg_daily_vol,
            'avg_weekly_volatility': avg_weekly_vol,
            'recommendation': '',
            'reason': '',
            'risk_level': '',
            'suggested_action': '',
            'target_price': None,
            'stop_loss': None
        }

        # 综合判断买入/卖出建议
        if price_trend > 3 and latest_daily_vol < avg_daily_vol * 1.5:
            # 上涨趋势，波动适中 - 可能考虑止盈
            recommendation['recommendation'] = '卖出/止盈'
            recommendation['reason'] = '股价处于上涨趋势，涨幅适中，建议考虑获利了结'
            recommendation['risk_level'] = '中等'
            recommendation['suggested_action'] = '可以考虑分批卖出，锁定收益'
            recommendation['target_price'] = round(latest_price * 0.95, 2)  # 回调5%考虑
            recommendation['stop_loss'] = round(latest_price * 1.02, 2)  # 突破2%止损

        elif price_trend < -3 and latest_daily_vol > avg_daily_vol * 1.5:
            # 下跌趋势，波动较大 - 可以考虑抄底
            recommendation['recommendation'] = '买入/抄底'
            recommendation['reason'] = '股价处于下跌趋势，但波动较大，可能存在反弹机会'
            recommendation['risk_level'] = '较高'
            recommendation['suggested_action'] = '可以分批建仓，控制仓位'
            recommendation['target_price'] = round(latest_price * 1.1, 2)  # 上涨10%目标
            recommendation['stop_loss'] = round(latest_price * 0.9, 2)  # 下跌10%止损

        elif latest_daily_vol < avg_daily_vol * 0.8:
            # 波动较小，价格稳定 - 可以考虑布局
            recommendation['recommendation'] = '观察/持有'
            recommendation['reason'] = '当前波动较小，价格稳定，建议继续观察'
            recommendation['risk_level'] = '较低'
            recommendation['suggested_action'] = '可以持有观望，等待更好的入场时机'

        elif latest_weekly_vol > avg_weekly_vol * 2:
            # 周波动过大 - 风险较高
            recommendation['recommendation'] = '谨慎/观望'
            recommendation['reason'] = '周波动过大，市场不确定性较高，建议谨慎操作'
            recommendation['risk_level'] = '高'
            recommendation['suggested_action'] = '建议观望，等待趋势明朗'

        else:
            recommendation['recommendation'] = '中性/持有'
            recommendation['reason'] = '当前无明显买卖信号，建议保持现有仓位'
            recommendation['risk_level'] = '中等'
            recommendation['suggested_action'] = '继续持有，关注市场变化'

        return recommendation

    def visualize_stock(self, stock_code: str, days: int = 30, save_path: str = None):
        """
        可视化股票波动

        Args:
            stock_code: 股票代码
            days: 天数
            save_path: 保存路径
        """
        print(f"\n正在获取股票 {stock_code} 的数据...")

        # 获取数据
        stock_data = self.get_stock_data(stock_code, days)

        if not stock_data or 'data' not in stock_data:
            print("无法获取股票数据")
            return

        data = stock_data['data']
        stock_name = stock_data.get('name', stock_code)

        print(f"成功获取 {len(data)} 天的数据")

        # 计算波动
        daily_vol, weekly_vol = self.calculate_volatility(data)

        # 生成交易建议
        recommendation = self.generate_trading_recommendation(data, daily_vol, weekly_vol)

        # 准备绘图数据
        dates = [datetime.strptime(item['date'], '%Y-%m-%d') for item in data]
        prices = [item['close'] for item in data]

        # 创建图表
        fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(14, 10))

        # 子图1：股价走势
        ax1.plot(dates, prices, 'b-', linewidth=2, label='收盘价')
        ax1.fill_between(dates, prices, alpha=0.3)
        ax1.set_title(f'{stock_name}({stock_code}) - 股价走势 (最近{days}天)', fontsize=14, fontweight='bold')
        ax1.set_xlabel('日期', fontsize=12)
        ax1.set_ylabel('价格', fontsize=12)
        ax1.grid(True, alpha=0.3)
        ax1.legend()

        # 格式化x轴日期
        ax1.xaxis.set_major_formatter(mdates.DateFormatter('%Y-%m-%d'))
        ax1.xaxis.set_major_locator(mdates.DayLocator(interval=5))
        plt.setp(ax1.xaxis.get_majorticklabels(), rotation=45, ha='right')

        # 子图2：波动分析
        ax2.plot(dates[:len(daily_vol)], daily_vol, 'r-', linewidth=2, label='日波动', alpha=0.7)
        ax2.plot(dates[:len(weekly_vol)], weekly_vol, 'g-', linewidth=2, label='周波动', alpha=0.7)
        ax2.fill_between(dates[:len(daily_vol)], daily_vol, alpha=0.2, color='red')
        ax2.fill_between(dates[:len(weekly_vol)], weekly_vol, alpha=0.2, color='green')
        ax2.set_title('波动分析', fontsize=14, fontweight='bold')
        ax2.set_xlabel('日期', fontsize=12)
        ax2.set_ylabel('波动率 (%)', fontsize=12)
        ax2.grid(True, alpha=0.3)
        ax2.legend()

        # 格式化x轴日期
        ax2.xaxis.set_major_formatter(mdates.DateFormatter('%Y-%m-%d'))
        ax2.xaxis.set_major_locator(mdates.DayLocator(interval=5))
        plt.setp(ax2.xaxis.get_majorticklabels(), rotation=45, ha='right')

        # 添加交易建议文本框
        recommendation_text = f"""
交易建议：
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
当前价格: {recommendation['current_price']:.2f}
价格趋势: {recommendation['price_trend']:+.2f}%
日波动率: {recommendation['daily_volatility']:.2f}%
周波动率: {recommendation['weekly_volatility']:.2f}%
平均日波动: {recommendation['avg_daily_volatility']:.2f}%
平均周波动: {recommendation['avg_weekly_volatility']:.2f}%

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
建议: {recommendation['recommendation']}
风险等级: {recommendation['risk_level']}
理由: {recommendation['reason']}

操作建议: {recommendation['suggested_action']}
"""
        if recommendation['target_price']:
            recommendation_text += f"目标价格: {recommendation['target_price']:.2f}\n"
        if recommendation['stop_loss']:
            recommendation_text += f"止损价格: {recommendation['stop_loss']:.2f}\n"

        plt.figtext(0.5, 0.02, recommendation_text, fontsize=9,
                    bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.5),
                    ha='center', va='bottom')

        plt.tight_layout(rect=[0, 0.3, 1, 1])

        # 保存或显示
        if save_path:
            plt.savefig(save_path, dpi=300, bbox_inches='tight')
            print(f"图表已保存到: {save_path}")
        else:
            default_path = f"./股票可视化_{stock_code}_{datetime.now().strftime('%Y%m%d')}.png"
            plt.savefig(default_path, dpi=300, bbox_inches='tight')
            print(f"图表已保存到: {default_path}")

        plt.show()

        # 打印交易建议
        print(f"\n{'='*60}")
        print(f"{stock_name}({stock_code}) 交易建议")
        print(f"{'='*60}")
        print(recommendation_text)

        return recommendation


def main():
    """
    主函数 - 演示股票可视化Skill的使用
    """
    print("="*60)
    print("股票可视化Skill")
    print("="*60)

    skill = StockVisualizationSkill()

    # 示例：分析某只股票
    stock_code = input("\n请输入股票代码 (默认: 000001): ").strip() or "000001"
    days = int(input("请输入分析天数 (默认: 30): ").strip() or "30")

    print(f"\n开始分析股票 {stock_code} 最近 {days} 天的数据...")
    print("="*60)

    try:
        recommendation = skill.visualize_stock(stock_code, days)

        # 保存交易建议到JSON文件
        output_file = f"./交易建议_{stock_code}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
        with open(output_file, 'w', encoding='utf-8') as f:
            json.dump(recommendation, f, ensure_ascii=False, indent=2)

        print(f"\n交易建议已保存到: {output_file}")

    except Exception as e:
        print(f"分析失败: {e}")
        import traceback
        traceback.print_exc()


if __name__ == "__main__":
    main()
