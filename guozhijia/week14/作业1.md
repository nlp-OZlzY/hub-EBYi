# -*- coding: utf-8 -*-
"""
作业1：基于LangChain的本地知识库问答系统
功能：文档检索 + LLM回答流程
"""

import os
from typing import List
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import TextLoader, PyPDFLoader
from langchain_community.vectorstores import FAISS
from langchain_community.embeddings import OllamaEmbeddings
from langchain.prompts import ChatPromptTemplate
from langchain_community.chat_models import ChatOllama
from langchain.chains import RetrievalQA

# 配置
EMBEDDING_MODEL = "nomic-embed-text"  # 使用Ollama的嵌入模型
LLM_MODEL = "qwen2.5"  # 使用Ollama的LLM模型
KNOWLEDGE_BASE_DIR = "./knowledge_base"  # 知识库目录


def load_documents(directory: str) -> List:
    """
    加载知识库中的所有文档
    支持文本文件和PDF文件
    """
    documents = []
    
    if not os.path.exists(directory):
        print(f"知识库目录 {directory} 不存在，将创建该目录")
        os.makedirs(directory)
        return documents
    
    for filename in os.listdir(directory):
        filepath = os.path.join(directory, filename)
        
        if filename.endswith('.txt'):
            loader = TextLoader(filepath, encoding='utf-8')
            docs = loader.load()
            documents.extend(docs)
        elif filename.endswith('.pdf'):
            loader = PyPDFLoader(filepath)
            docs = loader.load()
            documents.extend(docs)
    
    print(f"成功加载 {len(documents)} 个文档片段")
    return documents


def split_documents(documents: List) -> List:
    """
    将文档分割成较小的块
    """
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=500,
        chunk_overlap=50,
        length_function=len,
    )
    
    splits = text_splitter.split_documents(documents)
    print(f"文档分割为 {len(splits)} 个块")
    return splits


def create_vector_store(splits: List) -> FAISS:
    """
    创建向量存储
    """
    print("正在创建向量存储，这可能需要几分钟...")
    
    embeddings = OllamaEmbeddings(model=EMBEDDING_MODEL)
    vectorstore = FAISS.from_documents(splits, embeddings)
    
    print("向量存储创建完成")
    return vectorstore


def create_qa_chain(vectorstore: FAISS) -> RetrievalQA:
    """
    创建问答链
    """
    llm = ChatOllama(model=LLM_MODEL, temperature=0)
    
    retriever = vectorstore.as_retriever(
        search_type="similarity",
        search_kwargs={"k": 3}  # 返回最相关的3个文档块
    )
    
    # 创建自定义提示词模板
    template = """你是一个专业的知识库助手。请基于以下检索到的文档内容回答用户的问题。

检索到的文档内容：
{context}

用户问题：{question}

请：
1. 基于检索到的文档内容回答问题
2. 如果文档中没有相关信息，请明确说明
3. 回答要清晰、准确、有条理

回答："""
    
    prompt = ChatPromptTemplate.from_template(template)
    
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        chain_type="stuff",
        retriever=retriever,
        chain_type_kwargs={"prompt": prompt},
        return_source_documents=True
    )
    
    return qa_chain


def ask_question(qa_chain: RetrievalQA, question: str):
    """
    向知识库提问
    """
    print(f"\n{'='*60}")
    print(f"用户问题: {question}")
    print(f"{'='*60}")
    
    result = qa_chain({"query": question})
    
    print(f"\n回答:\n{result['result']}")
    
    print(f"\n{'='*60}")
    print("引用的文档来源:")
    print(f"{'='*60}")
    for i, doc in enumerate(result['source_documents'], 1):
        print(f"\n文档 {i}:")
        print(f"内容: {doc.page_content[:200]}...")
        print(f"元数据: {doc.metadata}")


def main():
    """
    主函数
    """
    print("="*60)
    print("LangChain 本地知识库问答系统")
    print("="*60)
    
    # 1. 加载文档
    print("\n步骤1: 加载文档...")
    documents = load_documents(KNOWLEDGE_BASE_DIR)
    
    if not documents:
        print("警告: 知识库中没有文档，请先添加文档到 knowledge_base 目录")
        print("提示: 你可以添加 .txt 或 .pdf 文件")
        return
    
    # 2. 分割文档
    print("\n步骤2: 分割文档...")
    splits = split_documents(documents)
    
    # 3. 创建向量存储
    print("\n步骤3: 创建向量存储...")
    vectorstore = create_vector_store(splits)
    
    # 4. 创建问答链
    print("\n步骤4: 创建问答链...")
    qa_chain = create_qa_chain(vectorstore)
    
    # 5. 交互式问答
    print("\n步骤5: 开始问答系统")
    print("输入 'exit' 或 'quit' 退出程序")
    print("="*60)
    
    while True:
        try:
            question = input("\n请输入你的问题: ").strip()
            
            if question.lower() in ['exit', 'quit', '退出']:
                print("感谢使用，再见！")
                break
            
            if not question:
                print("请输入有效的问题")
                continue
            
            ask_question(qa_chain, question)
            
        except KeyboardInterrupt:
            print("\n\n感谢使用，再见！")
            break
        except Exception as e:
            print(f"发生错误: {e}")


if __name__ == "__main__":
    main()
