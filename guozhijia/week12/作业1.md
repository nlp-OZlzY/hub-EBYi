"""
运行记录：基于 chinook.db 的 NL2SQL Agent 问答测试
使用本地 SQLite 直接验证 SQL 查询结果
"""

import sqlite3

DB_PATH = "./chinook.db"

def run_sql(sql: str):
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    cursor.execute(sql)
    result = cursor.fetchall()
    conn.close()
    return result


# ============== 提问1: 数据库中总共有多少张表 ==============
print("=" * 60)
print("提问1: 数据库中总共有多少张表？")
print("=" * 60)

sql1 = "SELECT COUNT(*) FROM sqlite_master WHERE type='table' AND name NOT LIKE 'sqlite_%';"
result1 = run_sql(sql1)
print(f"\n生成的 SQL:\n{sql1}")
print(f"\n查询结果: {result1}")
print(f"\n回答: 数据库中总共有 {result1[0][0]} 张表")

# 补充：列出所有表名
sql1_detail = "SELECT name FROM sqlite_master WHERE type='table' AND name NOT LIKE 'sqlite_%' ORDER BY name;"
tables = run_sql(sql1_detail)
print(f"\n所有表名列表:")
for t in tables:
    print(f"  - {t[0]}")
