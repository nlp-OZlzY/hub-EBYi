"""
使用云端Qwen-VL解析本地PDF第一页
"""

import os
import base64
import openai  # 使用openai库，兼容Qwen-VL API格式
import fitz  # PyMuPDF - 跨平台的PDF处理库

# Qwen-VL API配置
qwen_client = openai.OpenAI(
    api_key="sk-f988e27a2644492485d84f2526d4f83d",
    base_url="https://dashscope.aliyuncs.com/compatible-mode/v1"
)


def pdf_page_to_image(pdf_path: str, page_number: int = 1, zoom: float = 2.0) -> str:
    """
    将PDF指定页面转换为图片

    Args:
        pdf_path: PDF文件路径
        page_number: 页码（从1开始），默认第1页
        zoom: 缩放比例，默认2.0（即200%）

    Returns:
        生成的图片路径
    """
    try:
        # 打开PDF文档
        doc = fitz.open(pdf_path)

        # 检查页码是否有效
        if page_number < 1 or page_number > len(doc):
            print(f"页码 {page_number} 超出范围，文档共有 {len(doc)} 页")
            return None

        # 获取指定页面（PyMuPDF页码从0开始）
        page = doc[page_number - 1]

        # 计算缩放后的矩阵
        mat = fitz.Matrix(zoom, zoom)

        # 将页面渲染为图片
        pix = page.get_pixmap(matrix=mat)

        # 生成输出路径
        output_path = pdf_path.replace('.pdf', f'_page_{page_number}.png')

        # 保存图片
        pix.save(output_path)

        # 关闭文档
        doc.close()

        return output_path
    except Exception as e:
        print(f"PDF转图片失败: {e}")
        return None


def encode_image_to_base64(image_path: str) -> str:
    """将图片编码为base64字符串"""
    with open(image_path, "rb") as image_file:
        return base64.b64encode(image_file.read()).decode('utf-8')


def parse_pdf_first_page_with_qwen_vl(
    pdf_path: str,
    prompt: str = "请详细描述这张图片的内容，包括所有文字、图表、公式等。"
) -> str:
    """
    使用Qwen-VL解析PDF第一页

    Args:
        pdf_path: PDF文件路径
        prompt: 发送给模型的提示词

    Returns:
        Qwen-VL的解析结果
    """
    # 1. 将PDF第一页转换为图片
    image_path = pdf_page_to_image(pdf_path, page_number=1, zoom=2.0)

    if not image_path:
        return "PDF转图片失败"

    try:
        # 2. 读取图片并进行base64编码
        with open(image_path, "rb") as image_file:
            base64_image = base64.b64encode(image_file.read()).decode('utf-8')

        # 3. 调用Qwen-VL API
        response = qwen_client.chat.completions.create(
            model="qwen-vl-max",  # 使用Qwen-VL-Max模型
            messages=[
                {
                    "role": "user",
                    "content": [
                        {
                            "type": "image_url",
                            "image_url": {
                                "url": f"data:image/jpeg;base64,{base64_image}"
                            }
                        },
                        {
                            "type": "text",
                            "text": prompt
                        }
                    ]
                }
            ],
            max_tokens=2000
        )

        result = response.choices[0].message.content
        return result

    finally:
        # 4. 清理临时生成的图片
        if os.path.exists(image_path):
            os.remove(image_path)


def main():
    """主函数示例"""
    # PDF文件目录
    pdf_dir = "/Users/guozhijia/Documents/八斗/资料/推荐书籍/"

    # 选择一个PDF文件
    pdf_file = os.path.join(pdf_dir, "机器学习_周志华.pdf")

    if not os.path.exists(pdf_file):
        print(f"文件不存在: {pdf_file}")
        return

    print(f"正在解析: {pdf_file}")
    print("-" * 50)

    # 调用Qwen-VL解析
    result = parse_pdf_first_page_with_qwen_vl(
        pdf_path=pdf_file,
        prompt="请详细描述这张图片的内容，包括所有文字、图表、公式等。"
    )

    print("Qwen-VL解析结果:")
    print(result)


if __name__ == "__main__":
    main()
