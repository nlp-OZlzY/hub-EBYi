# 代码
```python
"""
Chinese CLIP Zero-Shot Image Classification
使用中文 CLIP 模型对小狗照片进行零样本图像分类
"""

import torch
from PIL import Image
from transformers import ChineseCLIPProcessor, ChineseCLIPModel

# 设置图片路径
image_path = "小狗照片.png"

# 定义候选类别（中文）
candidate_labels = [
    "狗的照片",
    "猫的照片",
    "鸟的照片",
    "人物照片",
    "汽车照片",
    "风景照片",
    "食物照片",
    "建筑物照片"
]

def main():
    # 加载 Chinese CLIP 模型和处理器 (使用 HuggingFace 提供的正确中文模型)
    print("Loading Chinese CLIP model...")
    model = ChineseCLIPModel.from_pretrained("OFA-Sys/chinese-clip-vit-base-patch16")
    processor = ChineseCLIPProcessor.from_pretrained("OFA-Sys/chinese-clip-vit-base-patch16")

    # 加载图片
    print(f"Loading image: {image_path}")
    image = Image.open(image_path)

    # 显示图片信息
    print(f"Image size: {image.size}")
    print(f"Image mode: {image.mode}")

    # 进行 zero-shot 分类
    print("\nRunning zero-shot classification with Chinese CLIP...")
    print("=" * 50)

    # 使用 processor 处理图片和文本
    inputs = processor(
        text=candidate_labels,
        images=image,
        return_tensors="pt",
        padding=True
    )

    # 获取模型输出
    with torch.no_grad():
        outputs = model(**inputs)

    # 计算相似度分数
    logits_per_image = outputs.logits_per_image  # 图片与文本的匹配分数
    probs = logits_per_image.softmax(dim=1)[0]  # 转换为概率

    # 排序并显示结果
    results = list(zip(candidate_labels, probs.tolist()))
    results.sort(key=lambda x: x[1], reverse=True)

    print("\n中文 CLIP 分类结果 (按概率排序):")
    print("-" * 50)
    for i, (label, prob) in enumerate(results, 1):
        bar = "█" * int(prob * 30)
        print(f"{i}. {label:10s}: {prob:.4f} {bar}")

    print("-" * 50)
    print(f"\n✓ 最可能的类别: {results[0][0]} (概率: {results[0][1]:.2%})")

if __name__ == "__main__":
    main()

```
